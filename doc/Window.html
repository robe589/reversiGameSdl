def initialize(backColor)
	@backColor=backColor
	SDL.init(SDL::INIT_VIDEO)
	SDL::TTF.init
	@screen = SDL::Screen.open(@@width,@@height, 16, SDL::SWSURFACE)
	SDL::WM::set_caption(@@windowName,'testsprite.rb icon')

	#背景色を赤で塗りつぶし
	@screen.fill_rect(0,0,@@width,@@height,@backColor)
end
